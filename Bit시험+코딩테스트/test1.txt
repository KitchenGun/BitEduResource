using System;

public class Solution 
{
    int day=0;
    int holidayCount;
    string[] days = new string[365];
    bool[] isHoliday= new bool[365];
    
    public int solution(string[] holidays, int k) 
    {
        //주말 휴일로 기록
        SetWeekend();
        //휴일 추가
        SetHoliday(holidays);
        //휴일수 세기
        bool isprevdayholi = false;
        foreach(bool isholiday in isHoliday)
        {
            if(isholiday == true)
            {
                isprevdayholi = false;
                //전날이 휴일이 아닐경우
                if(!isprevdayholi)
                {
                    isprevdayholi=true;
                }
            }
        }
        
        int answer = -1;
        return answer;
    }
    
    private void SetWeekend()
    {
        //주말 공휴일 처리
        for(int i=1;i<=365;i++)
        {
            if(i%7==2||i%7==3)
            {
                isHoliday[i]=true;
            }
            else
            {
                isHoliday[i]=false;
            }
        }
    }
    private void SetHoliday(string[] holidays)
    {
        foreach(string holiday in holidays)
        {
            string[] result = days.Split(new string[] {"/"}, StringSplitOptions.None);
            if(result[0]=="01")
            {
                day=Int32.Parse(result[1]);
                isHoliday[day]=true;
            }
            else if(result[0]=="02")
            {
                day=Int32.Parse(result[1]);
                isHoliday[day+31]=true;
            }
            else if(result[0]=="08")
            {
                day=Int32.Parse(result[1]);
                isHoliday[day+(31*6)+28]=true;
            }
            else if(Int32.Parse(result[1])%2==1)
            {
                day=Int32.Parse(result[1]);
                isHoliday[day+((31*(Int32.Parse(result[1])-1))+28)]=true;
            }
            else
            {
                day=Int32.Parse(result[1]);
                isHoliday[day+((30*(Int32.Parse(result[1])-1))+28)]=true;
            }
        }
    }
    private void CountHoliday()
    {
        //휴일수 세기
        List<int> holidayLengthlist = new List<int>();
 
        bool isprevdayholi = false;
        bool isnextdayholi = false;
        day=0;
        for(int i = 1;i<=365;i++)
        {
            //지금 휴일인 경우
            if(isHoliday[i]==true)
            {   //전날이 휴일인 경우
                if(isprevdayholi)
                {
                    day++;
                }
                else
                {
                    day=0;
                    day++;
                }
            }
            else
            {
                if(isprevdayholi)
                {
                    day++;
                }
            }
        }
    }
    
}
